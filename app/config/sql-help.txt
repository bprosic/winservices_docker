SELECT p.hostsName, p.hostsIpAddr, p.hostsServiceName, p.hostsServiceStatus, p.createdAt
FROM winservices.tblhosts AS p
INNER JOIN (
SELECT hostsIpAddr, MAX(idHosts) as id
FROM winservices.tblhosts
GROUP BY hostsIpAddr
)as last_ip
on p.idHosts = last_ip.idHosts
WHERE p.hostsIpAddr IN ('192.168.162.18', '192.168.162.13', '192.168.56.1')

SELECT p.hostsName, p.hostsIpAddr, p.hostsServiceName, p.hostsServiceStatus, p.createdAt
FROM winservices.tblhosts AS p
INNER JOIN (
SELECT hostsIpAddr, MAX(createdAt) as createdAt, MAX(idHosts)
FROM winservices.tblhosts
GROUP BY hostsIpAddr
)as last_ip
on p.hostsIpAddr = last_ip.hostsIpAddr
AND p.createdAt = last_ip.createdAt
WHERE p.hostsIpAddr IN ('192.168.162.18', '192.168.162.13', '192.168.56.1')
